CREATE TABLE SPORTS (
    event_id INT NOT NULL AUTO_INCREMENT,
    venue_id INT NOT NULL,
    game_name VARCHAR(150) NOT NULL,
    sport_name VARCHAR(150) NOT NULL,
    date DATE NOT NULL,
    time TIME NOT NULL,
    CONSTRAINT sport_pk PRIMARY KEY (event_id),
    CONSTRAINT sport_venue_fk FOREIGN KEY(event_id) REFERENCES venue(venue_id)
);

CREATE TABLE CONCERT (
    event_id INT NOT NULL AUTO_INCREMENT,
    concert_name VARCHAR(150) NOT NULL,
    date DATE NOT NULL,
    time TIME NOT NULL,
    CONSTRAINT concert_pk PRIMARY KEY (event_id)
);
